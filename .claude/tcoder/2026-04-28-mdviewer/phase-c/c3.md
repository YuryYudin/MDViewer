# C3: Share/export dialog and migrate-sidecars CLI subcommand

**Avoid:**
- Do NOT zip the export. WHY: the design's Key Decision rejects zip bundles in favor of a plain folder containing `<doc>.md` + `<doc>.md.comments.json` to keep the receiver's "open in any tool" workflow intact. A zip forces them to extract first.
- Do NOT migrate sidecars in place without an opt-in flag if you decide to add one. WHY: this CLI subcommand is invoked explicitly by the user; adding a `--dry-run` for safety is fine but the *default* behavior must do the migration the user asked for. The design says it walks the directory and rewrites — match that.
- Do NOT export the v1 shape from C3 once C1 has migrated to v2. WHY: a counterpart on the latest version expects v2; an exported v1 would force a re-migration on their side and erase any comments they added in the meantime. Always export the current schema_version.
- Do NOT silently overwrite an existing export folder. WHY: the user could lose unrelated files. Detect the case and fail with a clear error so the Share dialog can prompt for a different folder.

### Step 2: Implement ShareDialog.ts

Create `src/views/ShareDialog.ts` via `createElement` so the path strings go through `textContent`:

```ts
import type { Ipc } from '../ipc';

export async function mountShareDialog(
  root: HTMLElement,
  ipc: Ipc & { exportDocument(args: { tabId: string; folder: string }): Promise<{ folder: string; files: string[] }> },
  args: { tabId: string; path: string },
): Promise<void> {
  root.replaceChildren();
  const view = document.createElement('section');
  view.setAttribute('data-view', 'share');

  const explainer = document.createElement('p');
  explainer.textContent = 'MDViewer will export the markdown plus its comments sidecar to a folder you pick. Send both files to your reviewer.';
  view.appendChild(explainer);

  const list = document.createElement('ul');
  for (const name of [args.path.split('/').pop()!, args.path.split('/').pop()! + '.comments.json']) {
    const li = document.createElement('li');
    li.setAttribute('data-test', 'preview-name');
    li.textContent = name;
    list.appendChild(li);
  }
  view.appendChild(list);

  const folder = document.createElement('input');
  folder.setAttribute('data-test', 'folder');
  folder.placeholder = 'Choose a folder…';
  view.appendChild(folder);

  const cancel = document.createElement('button');
  cancel.setAttribute('data-action', 'cancel'); cancel.textContent = 'Cancel';
  cancel.addEventListener('click', () => view.dispatchEvent(new CustomEvent('share-dismissed', { bubbles: true })));

  const exp = document.createElement('button');
  exp.setAttribute('data-action', 'export'); exp.textContent = 'Export';
  exp.addEventListener('click', async () => {
    const r = await ipc.exportDocument({ tabId: args.tabId, folder: folder.value });
    view.dispatchEvent(new CustomEvent('share-exported', { bubbles: true, detail: r }));
  });

  view.append(cancel, exp);
  root.appendChild(view);
}
```

Then modify `src-tauri/src/main.rs`:

```rust
#[tauri::command]
fn export_document(state: State<'_, Ws>, tab_id: String, folder: PathBuf) -> Result<ExportResult, String> {
    let ws = state.lock().map_err(|e| e.to_string())?;
    let tab = ws.tab(&tab_id).ok_or_else(|| "no such tab".to_string())?;
    if folder.exists() && std::fs::read_dir(&folder).map(|d| d.count()).unwrap_or(0) > 0 {
        return Err("export folder is not empty".into());
    }
    std::fs::create_dir_all(&folder).map_err(|e| e.to_string())?;
    let md_dest = folder.join(tab.path.file_name().unwrap());
    std::fs::copy(&tab.path, &md_dest).map_err(|e| e.to_string())?;

    let pattern = ws.settings_store().get().comments.sidecar_pattern.clone();
    let local_sc = mdviewer_lib::sidecar::sidecar_path(&tab.path, &pattern);
    if local_sc.exists() {
        let sc_dest = folder.join(local_sc.file_name().unwrap());
        std::fs::copy(&local_sc, &sc_dest).map_err(|e| e.to_string())?;
    }
    Ok(ExportResult {
        folder: folder.clone(),
        files: vec![md_dest.file_name().unwrap().to_string_lossy().into_owned(),
                    local_sc.file_name().unwrap().to_string_lossy().into_owned()],
    })
}
```

(`ExportResult` is a `Serialize` struct with `folder: PathBuf` and `files: Vec<String>`.)

### Step 4: Implement migrate-sidecars CLI subcommand

Modify `src-tauri/src/main.rs`:

```rust
fn main() {
    let args: Vec<String> = std::env::args().collect();
    if args.len() >= 3 && args[1] == "migrate-sidecars" {
        let dir = PathBuf::from(&args[2]);
        if let Err(e) = migrate_sidecars(&dir) {
            eprintln!("migrate-sidecars failed: {e}");
            std::process::exit(1);
        }
        return;
    }
    // ... existing tauri::Builder...
}

fn migrate_sidecars(dir: &Path) -> anyhow::Result<()> {
    let mut migrated = 0usize;
    let mut already_v2 = 0usize;
    for entry in walkdir::WalkDir::new(dir) {
        let entry = entry?;
        if !entry.file_type().is_file() { continue; }
        let p = entry.path();
        let name = p.file_name().and_then(|s| s.to_str()).unwrap_or("");
        if !name.ends_with(".comments.json") { continue; }

        let bytes = std::fs::read(p)?;
        // Try v2 envelope first.
        if let Ok(env) = serde_json::from_slice::<serde_json::Value>(&bytes) {
            if env.get("schema_version").and_then(|v| v.as_u64()) == Some(2) {
                already_v2 += 1; continue;
            }
        }
        // v1 → load via sidecar.rs (which migrates), then save.
        let store = mdviewer_lib::sidecar::load_sidecar(p)?;
        mdviewer_lib::sidecar::save_sidecar(p, &store)?;
        migrated += 1;
        println!("migrated: {}", p.display());
    }
    println!("Done. migrated={migrated}, already_v2={already_v2}");
    Ok(())
}
```

Modify `src-tauri/Cargo.toml`:

```toml

walkdir = "2"
```
