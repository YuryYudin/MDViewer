# A11: Settings view (full panel)

**Avoid:**
- Do NOT enable the Sync provider control. WHY: the design lists it as deferred and the success-criterion-9 wording calls out that the Sync control is intentionally inert. Rendering it active would imply the feature ships and confuse users.
- Do NOT make shortcut bindings editable. WHY: Non-Goals explicitly defers interactive remap. Show the shortcuts table in read-only form so the schema A3 introduced is visible without committing to capture-keystroke UX.
- Do NOT round-trip every keystroke through `set_settings`. WHY: settings IPC writes to TOML on every call; typing a 12-char display name would emit 12 disk writes. Debounce free-text fields locally for ~250ms and commit on blur.
- Do NOT read MDVIEWER_VERSION from `npm` package metadata. WHY: the design says the About panel reads build-time constants emitted by `src-tauri/build.rs` and re-exported via `app_info` IPC. Reading from a package.json field would skip the commit hash and drift.

### Step 2: Implement Settings.ts

Create `src/views/Settings.ts`. Build the seven sections via `createElement` helpers — each section is purely structural and any user data (display name, sidecar pattern) goes into `value` properties or `textContent`, never into raw markup. Skeleton:

```ts
import type { Ipc, Settings } from '../ipc';

export async function mountSettings(root: HTMLElement, ipc: Ipc): Promise<void> {
  const settings = await ipc.getSettings();
  const info = await ipc.appInfo();
  root.replaceChildren();

  const view = document.createElement('section');
  view.setAttribute('data-view', 'settings');
  view.appendChild(buildProfile(ipc, settings));
  view.appendChild(buildAppearance(ipc, settings));
  view.appendChild(buildEditor(ipc, settings));
  view.appendChild(buildComments(ipc, settings));
  view.appendChild(buildShortcuts(settings));
  view.appendChild(buildAdvanced(ipc, settings));
  view.appendChild(buildAbout(info));
  root.appendChild(view);
}

function section(id: string): HTMLElement {
  const s = document.createElement('section');
  s.setAttribute('data-section', id);
  s.id = id;
  return s;
}

function labeledInput(label: string, attrs: Record<string, string>): { label: HTMLLabelElement; input: HTMLInputElement } {
  const lbl = document.createElement('label');
  lbl.textContent = label + ' ';
  const inp = document.createElement('input');
  for (const [k, v] of Object.entries(attrs)) inp.setAttribute(k, v);
  lbl.appendChild(inp);
  return { label: lbl, input: inp };
}

function debounce<T extends (...args: any[]) => void>(fn: T, ms: number): T {
  let h: number | undefined;
  return ((...args: any[]) => {
    if (h) window.clearTimeout(h);
    h = window.setTimeout(() => fn(...args), ms) as unknown as number;
  }) as T;
}

function buildProfile(ipc: Ipc, settings: Settings): HTMLElement {
  const s = section('profile');
  const name = labeledInput('Display name', { 'data-test': 'profile-name', value: settings.profile.display_name });
  const color = labeledInput('Color', { type: 'color', 'data-test': 'profile-color', value: settings.profile.color });
  s.append(name.label, color.label);
  const save = debounce(() => {
    settings.profile.display_name = name.input.value;
    settings.profile.color = color.input.value;
    ipc.setSettings(settings);
  }, 250);
  name.input.addEventListener('input', save);
  color.input.addEventListener('input', save);
  name.input.addEventListener('blur', () => ipc.setSettings(settings));
  return s;
}

function buildAppearance(ipc: Ipc, settings: Settings): HTMLElement {
  const s = section('appearance');
  const themeLabel = document.createElement('label'); themeLabel.textContent = 'Theme ';
  const select = document.createElement('select'); select.setAttribute('data-test', 'theme-select');
  for (const [val, txt] of [['light', 'Light'], ['dark', 'Dark'], ['follow_system', 'Follow system']] as const) {
    const opt = document.createElement('option'); opt.value = val; opt.textContent = txt;
    if (settings.appearance.theme === val) opt.selected = true;
    select.appendChild(opt);
  }
  themeLabel.appendChild(select);
  s.appendChild(themeLabel);
  select.addEventListener('change', () => {
    settings.appearance.theme = select.value as Settings['appearance']['theme'];
    document.body.classList.toggle('theme-dark', settings.appearance.theme === 'dark');
    ipc.setSettings(settings);
  });

  const font = labeledInput('Font size', {
    type: 'number', 'data-test': 'font-size', value: String(settings.appearance.font_size_px),
  });
  s.appendChild(font.label);
  font.input.addEventListener('change', () => {
    settings.appearance.font_size_px = parseInt(font.input.value, 10);
    document.documentElement.style.setProperty('--font-size', `${settings.appearance.font_size_px}px`);
    ipc.setSettings(settings);
  });
  return s;
}

function buildEditor(_ipc: Ipc, _s: Settings): HTMLElement { return section('editor'); /* analogous */ }
function buildComments(_ipc: Ipc, _s: Settings): HTMLElement { return section('comments'); /* analogous */ }

function buildShortcuts(settings: Settings): HTMLElement {
  const s = section('shortcuts');
  const table = document.createElement('table');
  for (const [k, v] of Object.entries(settings.shortcuts)) {
    const tr = document.createElement('tr');
    const td1 = document.createElement('td'); td1.textContent = k;
    const td2 = document.createElement('td');
    const kbd = document.createElement('kbd'); kbd.textContent = v;
    td2.appendChild(kbd);
    tr.append(td1, td2); table.appendChild(tr);
  }
  s.appendChild(table);
  return s;
}

function buildAdvanced(ipc: Ipc, settings: Settings): HTMLElement {
  const s = section('advanced');
  const lbl = document.createElement('label'); lbl.textContent = 'Sync provider ';
  const sync = document.createElement('select');
  sync.setAttribute('data-test', 'sync-provider');
  sync.disabled = true;
  const opt = document.createElement('option');
  opt.textContent = 'None — cloud sync ships in a later release';
  sync.appendChild(opt);
  lbl.appendChild(sync); s.appendChild(lbl);

  const verbose = document.createElement('label'); verbose.textContent = 'Verbose logs';
  const cb = document.createElement('input'); cb.type = 'checkbox';
  cb.setAttribute('data-test', 'verbose-logs'); cb.checked = settings.advanced.verbose_logs;
  cb.addEventListener('change', () => { settings.advanced.verbose_logs = cb.checked; ipc.setSettings(settings); });
  verbose.prepend(cb); s.appendChild(verbose);
  return s;
}

function buildAbout(info: { version: string; commit_hash: string }): HTMLElement {
  const s = section('about');
  const p = document.createElement('p');
  p.textContent = `MDViewer ${info.version} (${info.commit_hash})`;
  s.appendChild(p);
  return s;
}
```
