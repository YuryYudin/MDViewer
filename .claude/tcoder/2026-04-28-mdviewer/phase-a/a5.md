# A5: Markdown document parser with data-src-offset annotations

**Avoid:**
- Do NOT translate DOM Range to source offsets in the frontend by walking text-content lengths. WHY: the design's data flow names `data-src-offset` HTML attributes as the authoritative bridge — character-counting walks would silently break around code blocks, escape sequences, and CRLF normalization, and would force a Rust-side reverse map that the design explicitly avoided. Emit the offsets at parse time and let the frontend read them.
- Do NOT skip the `mermaid_enabled` and `syntax_highlighting` settings here. WHY: those toggles are routed to this module per the Settings → consumer mapping; if they default-on at parse time and only the frontend respects them, the emitted HTML carries Mermaid `<div>` shells and syntect-coloured spans even when the user disabled them, and the Settings view will not match its wireframe behavior.
- Do NOT use `pulldown-cmark` defaults — the design names GFM tables, task-lists, and autolinks as success-criterion-1 features. WHY: the default parser disables these. Enable them via `Options::ENABLE_TABLES | Options::ENABLE_TASKLISTS | Options::ENABLE_GFM_AUTOLINK` (or the equivalent constants for the chosen pulldown-cmark version).
- Do NOT call `syntect` for every render. WHY: building a `SyntaxSet` and `ThemeSet` is multi-millisecond work; on a large document with many code blocks the per-tab open path will visibly lag. Cache them in a `OnceLock<SyntaxSet>` / `OnceLock<ThemeSet>` so first render pays the cost and subsequent renders amortize.

```rust
    // (start, end) covers a slice that, when concatenated by document order
```

### Step 2: Implement document.rs

Create `src-tauri/src/document.rs`:

```rust
use pulldown_cmark::{Event, Options, Parser, Tag, TagEnd, CodeBlockKind};
use std::sync::OnceLock;
use syntect::easy::HighlightLines;
use syntect::highlighting::{Theme, ThemeSet};
use syntect::parsing::SyntaxSet;
use syntect::util::{as_24_bit_terminal_escaped, LinesWithEndings};
use syntect::html::{styled_line_to_highlighted_html, IncludeBackground};

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize, ts_rs::TS)]
#[ts(export)]
pub struct RenderOptions {
    pub syntax_highlighting: bool,
    pub mermaid_enabled: bool,
}

impl Default for RenderOptions {
    fn default() -> Self {
        Self { syntax_highlighting: true, mermaid_enabled: true }
    }
}

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize, ts_rs::TS)]
#[ts(export)]
pub struct RenderResult {
    pub html: String,
    /// All (start, end) source offsets emitted as `data-src-offset` on inline
    /// elements. Useful for tests asserting round-trip integrity. The frontend
    /// does not consume this field; mark it `#[serde(default)]` so future
    /// schema-trim changes don't break the IPC payload.
    #[serde(default)]
    pub text_spans: Vec<(usize, usize)>,
}

fn syntax_set() -> &'static SyntaxSet {
    static SS: OnceLock<SyntaxSet> = OnceLock::new();
    SS.get_or_init(SyntaxSet::load_defaults_newlines)
}

fn theme() -> &'static Theme {
    static TS: OnceLock<ThemeSet> = OnceLock::new();
    let set = TS.get_or_init(ThemeSet::load_defaults);
    set.themes.get("InspiredGitHub").or_else(|| set.themes.values().next()).unwrap()
}

pub fn render_markdown(source: &str, opts: &RenderOptions) -> RenderResult {
    let mut options = Options::empty();
    options.insert(Options::ENABLE_TABLES);
    options.insert(Options::ENABLE_TASKLISTS);
    options.insert(Options::ENABLE_STRIKETHROUGH);
    options.insert(Options::ENABLE_GFM);

    let parser = Parser::new_ext(source, options).into_offset_iter();

    let mut html = String::with_capacity(source.len() * 2);
    let mut text_spans = Vec::new();
    let mut code_buffer = String::new();
    let mut code_lang: Option<String> = None;
    let mut in_code = false;

    for (event, range) in parser {
        match event {
            Event::Start(Tag::CodeBlock(CodeBlockKind::Fenced(lang))) => {
                in_code = true;
                code_lang = Some(lang.into_string());
                code_buffer.clear();
            }
            Event::End(TagEnd::CodeBlock) => {
                let lang = code_lang.take().unwrap_or_default();
                emit_code_block(&mut html, &code_buffer, &lang, opts);
                in_code = false;
            }
            Event::Text(t) if in_code => code_buffer.push_str(&t),
            Event::Text(t) => {
                let s = range.start;
                let e = range.end;
                text_spans.push((s, e));
                html.push_str(&format!(
                    "<span data-src-offset=\"{s}\" data-src-end=\"{e}\">{}</span>",
                    escape_html(&t)
                ));
            }
            Event::Start(tag) => emit_open(&mut html, &tag, range.start),
            Event::End(tag) => emit_close(&mut html, &tag),
            Event::SoftBreak => html.push('\n'),
            Event::HardBreak => html.push_str("<br/>"),
            Event::Code(c) => {
                let s = range.start;
                let e = range.end;
                text_spans.push((s, e));
                html.push_str(&format!(
                    "<code data-src-offset=\"{s}\" data-src-end=\"{e}\">{}</code>",
                    escape_html(&c)
                ));
            }
            _ => { /* lists, html, rules — emit minimally */ }
        }
    }

    RenderResult { html, text_spans }
}

fn emit_code_block(out: &mut String, code: &str, lang: &str, opts: &RenderOptions) {
    if lang == "mermaid" {
        if opts.mermaid_enabled {
            out.push_str(&format!("<div class=\"mermaid\">{}</div>", escape_html(code)));
            return;
        }
        // fall through to raw <pre><code>
    }
    if opts.syntax_highlighting && !lang.is_empty() && lang != "mermaid" {
        let ss = syntax_set();
        if let Some(syntax) = ss.find_syntax_by_token(lang).or_else(|| ss.find_syntax_by_extension(lang)) {
            let mut h = HighlightLines::new(syntax, theme());
            out.push_str("<pre><code class=\"hl\">");
            for line in LinesWithEndings::from(code) {
                if let Ok(regions) = h.highlight_line(line, ss) {
                    let html_line = styled_line_to_highlighted_html(&regions[..], IncludeBackground::No)
                        .unwrap_or_default();
                    out.push_str(&html_line);
                }
            }
            out.push_str("</code></pre>");
            return;
        }
    }
    out.push_str("<pre><code>");
    out.push_str(&escape_html(code));
    out.push_str("</code></pre>");
}

fn emit_open(out: &mut String, tag: &Tag, _start: usize) {
    match tag {
        Tag::Paragraph => out.push_str("<p>"),
        Tag::Heading { level, .. } => out.push_str(&format!("<h{}>", *level as u8)),
        Tag::BlockQuote(_) => out.push_str("<blockquote>"),
        Tag::List(Some(_)) => out.push_str("<ol>"),
        Tag::List(None) => out.push_str("<ul>"),
        Tag::Item => out.push_str("<li>"),
        Tag::Emphasis => out.push_str("<em>"),
        Tag::Strong => out.push_str("<strong>"),
        Tag::Strikethrough => out.push_str("<del>"),
        Tag::Link { dest_url, .. } => out.push_str(&format!("<a href=\"{}\">", escape_html(dest_url))),
        Tag::Image { dest_url, .. } => out.push_str(&format!("<img src=\"{}\" alt=\"", escape_html(dest_url))),
        Tag::Table(_) => out.push_str("<table>"),
        Tag::TableHead => out.push_str("<thead><tr>"),
        Tag::TableRow => out.push_str("<tr>"),
        Tag::TableCell => out.push_str("<td>"),
        _ => {}
    }
}

fn emit_close(out: &mut String, tag: &TagEnd) {
    match tag {
        TagEnd::Paragraph => out.push_str("</p>"),
        TagEnd::Heading(level) => out.push_str(&format!("</h{}>", *level as u8)),
        TagEnd::BlockQuote(_) => out.push_str("</blockquote>"),
        TagEnd::List(true) => out.push_str("</ol>"),
        TagEnd::List(false) => out.push_str("</ul>"),
        TagEnd::Item => out.push_str("</li>"),
        TagEnd::Emphasis => out.push_str("</em>"),
        TagEnd::Strong => out.push_str("</strong>"),
        TagEnd::Strikethrough => out.push_str("</del>"),
        TagEnd::Link => out.push_str("</a>"),
        TagEnd::Image => out.push_str("\"/>"),
        TagEnd::Table => out.push_str("</table>"),
        TagEnd::TableHead => out.push_str("</tr></thead>"),
        TagEnd::TableRow => out.push_str("</tr>"),
        TagEnd::TableCell => out.push_str("</td>"),
        _ => {}
    }
}

fn escape_html(s: &str) -> String {
    s.replace('&', "&amp;").replace('<', "&lt;").replace('>', "&gt;").replace('"', "&quot;")
}
```
