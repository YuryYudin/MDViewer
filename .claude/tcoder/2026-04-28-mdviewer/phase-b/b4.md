# B4: Edit view, mode toggle, orphan-comments UX

**Avoid:**
- Do NOT debounce in Rust. WHY: the design's data flow says the keystroke stream lives in the WebView; only the write decision crosses the boundary. Frontend debounce keeps IPC traffic bounded; Rust-side debounce would still receive every keystroke event, defeating the purpose.
- Do NOT silently delete orphan comments on mode switch. WHY: success-criterion-4 plus the orphan-comments UX wireframe (09) require Relocate / Keep / Delete actions. Auto-deletion would discard user content without confirmation — exactly the failure mode the wireframe was designed to prevent.
- Do NOT use a `<div contenteditable>` for the editor. WHY: the design says raw `.md` textarea — contenteditable normalizes whitespace, swallows tabs, and would diverge from the saved bytes. A `<textarea>` with `white-space: pre` is what wireframe 07 shows.
- Do NOT recompute `resolve_anchor` for every thread on every keystroke. WHY: the editor would lag on large documents. Only re-resolve on mode-toggle back to View, after `save_document` has flushed the new bytes.

### Step 2: Implement Edit.ts

Create `src/views/Edit.ts`:

```ts
import type { Ipc } from '../ipc';

export interface EditMountArgs {
  tabId: string;
  path: string;
  source: string;
  autoSave: boolean;
  autoSaveDebounceMs: number;
  wordWrap: boolean;
  showWhitespace: boolean;
}

export interface EditView {
  currentSource(): string;
  destroy(): void;
}

export function mountEdit(root: HTMLElement, ipc: Ipc & { saveDocument(path: string, contents: string): Promise<void> }, args: EditMountArgs): EditView {
  root.replaceChildren();
  const view = document.createElement('div');
  view.setAttribute('data-view', 'edit');
  const ta = document.createElement('textarea');
  ta.setAttribute('data-test', 'editor');
  ta.value = args.source;
  ta.style.whiteSpace = args.wordWrap ? 'pre-wrap' : 'pre';
  if (args.showWhitespace) ta.classList.add('show-whitespace');
  view.appendChild(ta);
  root.appendChild(view);

  let timer: number | undefined;
  function schedule() {
    if (!args.autoSave) return;
    if (timer) window.clearTimeout(timer);
    timer = window.setTimeout(() => {
      ipc.saveDocument(args.path, ta.value);
    }, args.autoSaveDebounceMs) as unknown as number;
  }
  ta.addEventListener('input', schedule);

  return {
    currentSource: () => ta.value,
    destroy: () => { if (timer) window.clearTimeout(timer); ta.remove(); },
  };
}
```

### Step 3: Implement OrphanComments.ts

Create `src/views/OrphanComments.ts` rendering wireframe 09 with `[data-action="relocate"]`, `[data-action="keep"]`, `[data-action="delete"]`. Build via `createElement` so author names and bodies go through `textContent`.

```ts
export interface OrphanItem {
  id: string;
  anchor: { exact: string };
  comments: { author: string; body: string }[];
}

export interface OrphanCallbacks {
  orphans: OrphanItem[];
  onRelocate(id: string): void;
  onKeep(id: string): void;
  onDelete(id: string): void;
}

export function mountOrphanComments(root: HTMLElement, opts: OrphanCallbacks): void {
  root.replaceChildren();
  const wrap = document.createElement('section');
  wrap.setAttribute('data-view', 'orphan-comments');
  for (const o of opts.orphans) {
    const item = document.createElement('article');
    item.setAttribute('data-orphan-id', o.id);
    const quote = document.createElement('p'); quote.className = 'quote';
    quote.textContent = `“${o.anchor.exact}”`;
    item.appendChild(quote);
    const body = document.createElement('p');
    body.textContent = o.comments[0]?.body ?? '';
    item.appendChild(body);
    for (const action of ['relocate', 'keep', 'delete'] as const) {
      const btn = document.createElement('button');
      btn.setAttribute('data-action', action);
      btn.textContent = action[0].toUpperCase() + action.slice(1);
      btn.addEventListener('click', () => {
        if (action === 'relocate') opts.onRelocate(o.id);
        else if (action === 'keep') opts.onKeep(o.id);
        else opts.onDelete(o.id);
      });
      item.appendChild(btn);
    }
    wrap.appendChild(item);
  }
  root.appendChild(wrap);
}
```
